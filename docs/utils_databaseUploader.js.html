/**
 * @file databaseUploader.js
 * @description This script reads data from a JSON file and uploads it to a MySQL database.
 * It is used for restoration purposes in case of database corruption, utilizing daily scrapes as backups if incremental backups fail.
 */

const mysql = require("mysql2"); // MySQL client for Node.js
const path = require("path"); // Module for handling file paths

const dayjs = require("dayjs"); // Library for parsing and formatting dates

/**
 * Checks if a record with the given ID already exists in the database.
 *
 * @param {number|string} ID - The ID of the record to check.
 * @returns {Promise<boolean>} - Resolves to true if the record exists, false otherwise.
 */
const checkIfExists = (ID) => {
  return new Promise((resolve, reject) => {
    const query = 'SELECT 1 FROM annonce WHERE ID = ? LIMIT 1';
    connection.query(query, [ID], (err, results) => {
      if (err) {
        return reject(err); // Reject the promise if there's an error
      }
      resolve(results.length > 0); // Resolve with true if record exists
    });
  });
};

/**
 * Inserts a data record into the 'annonce' table of the database.
 *
 * @param {Object} data - The data object containing fields to insert.
 * @returns {Promise<void>} - Resolves when the data is successfully inserted.
 */
const insertData = (data) => {
  const {
    ID,
    TITLE,
    BODY,
    REGION_ID,
    TIMESTAMP,
    CHECKSUM,
    URL,
    CVR,
    Homepage,
    possible_duplicate,
    body_hash,
    companyUrlFromAnnonce,
  } = data;

  // Format the TIMESTAMP field to 'YYYY-MM-DD HH:mm:ss' format
  const formattedTimestamp = TIMESTAMP ? dayjs(TIMESTAMP).format('YYYY-MM-DD HH:mm:ss') : null;

  // Ensure CVR is a valid integer; otherwise, set to null
  const formattedCVR = /^\d+$/.test(CVR) ? parseInt(CVR, 10) : null;

  // Ensure possible_duplicate is a valid integer; otherwise, set to null
  const formattedPossibleDuplicate = /^\d+$/.test(possible_duplicate) ? parseInt(possible_duplicate, 10) : null;

  // SQL query to insert data into the 'annonce' table
  const query = `
    INSERT INTO annonce (
      ID, TITLE, BODY, REGION_ID, TIMESTAMP, CHECKSUM, URL, CVR,
      Homepage, possible_duplicate, body_hash, companyurlfromannonce
    ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
  `;

  return new Promise((resolve, reject) => {
    connection.query(query, [
      ID,
      TITLE,
      BODY,
      REGION_ID,
      formattedTimestamp,         // Use the formatted timestamp
      CHECKSUM,
      URL,
      formattedCVR,               // Use the formatted CVR
      Homepage,
      formattedPossibleDuplicate, // Use the formatted possible_duplicate
      body_hash,
      companyUrlFromAnnonce,
    ], (err) => {
      if (err) {
        return reject(err); // Reject the promise if there's an error
      }
      resolve(); // Resolve the promise on successful insertion
    });
  });
};

/**
 * Counts the total number of lines in the JSON file to set the progress bar total.
 *
 * @param {string} filePath - The path to the JSON file.
 * @returns {Promise<number>} - Resolves with the total number of lines.
 */
const countTotalLines = (filePath) => {
  return new Promise((resolve, reject) => {
    let lineCount = 0;
    const readStream = fs.createReadStream(filePath, { encoding: 'utf8' });
    readStream.on('data', (chunk) => {
      lineCount += chunk.split('\n').length - 1; // Increment line count
    });
    readStream.on('end', () => resolve(lineCount)); // Resolve with the total line count
    readStream.on('error', reject); // Reject if there's an error reading the file
  });
};

/**
 * Processes the JSON file line by line, checks for existing records, and inserts new data into the database.
 *
 * @async
 * @function processJsonFile
 * @returns {Promise<void>}
 */
const processJsonFile = async () => {
  try {
    const totalLines = await countTotalLines(jsonFilePath); // Get total number of lines
    let processedLines = 0; // Counter for processed lines
    let statusMessage = 'Starting...'; // Status message for the progress bar

    // Start the progress bar
    progressBar.start(totalLines, 0, { status: statusMessage });

    const readStream = fs.createReadStream(jsonFilePath, { encoding: 'utf8' });
    let buffer = '';
    const promises = [];

    // Event listener for when data is received from the read stream
    readStream.on('data', (chunk) => {
      buffer += chunk;
      let boundary = buffer.indexOf('\n');

      // Process each line in the buffer
      while (boundary !== -1) {
        const line = buffer.substring(0, boundary);
        buffer = buffer.substring(boundary + 1);
        boundary = buffer.indexOf('\n');

        processedLines++; // Increment the processed lines counter
        try {
          const data = JSON.parse(line); // Parse the JSON line

          // Check if the record already exists in the database
          const promise = checkIfExists(data.ID)
            .then((exists) => {
              if (!exists) {
                // Insert data if it doesn't exist
                return insertData(data).then(() => {
                  statusMessage = `Inserted ID ${data.ID}`;
                  updateProgress();
                });
              } else {
                statusMessage = `ID ${data.ID} already exists. Skipping...`;
                updateProgress();
              }
            })
            .catch((err) => {
              statusMessage = `Error checking or inserting ID ${data.ID}: ${err.message}`;
              updateProgress();
            });

          promises.push(promise); // Add the promise to the array for later synchronization
        } catch (err) {
          statusMessage = `Error parsing JSON line: ${err.message}`;
          updateProgress();
        }
      }
    });

    // Event listener for when the read stream ends
    readStream.on('end', async () => {
      // Wait for all insert operations to complete
      await Promise.all(promises);
      progressBar.stop(); // Stop the progress bar
      console.log('Finished processing the JSON file.');
      connection.end(); // Close the database connection
    });

    // Event listener for read stream errors
    readStream.on('error', (err) => {
      statusMessage = `Error reading the JSON file: ${err.message}`;
      updateProgress();
    });

    /**
     * Updates the progress bar with the current status and processed lines.
     */
    const updateProgress = () => {
      progressBar.update(processedLines, { status: statusMessage });
    };
  } catch (err) {
    console.error('Error processing the JSON file:', err);
    connection.end(); // Close the database connection on error
  }
};

// Start processing the JSON file
processJsonFile();
